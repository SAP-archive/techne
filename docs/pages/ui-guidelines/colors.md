    <h2>Function</h2>
    <h3><code>@function tn-color($type: text, $shade: 1)</code></h3>

    <p>Example:<br>
    </p><pre>.foo {
        color: tn-color(action,2);
        background-color: tn-color(neutral);
    }
    </pre>
